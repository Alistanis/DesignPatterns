def execute_next
  if @next == 0
    @commands[@next].execute
    @next = @next + 1
  else
    if @commands[@next] != nil
      @commands[@next].execute
      @next = @next + 1
    else
      raise CommandList, 'No next command in list; index out of bounds'
    end
  end
end

def undo_last
  if @next > 0
    @commands[@next - 1].undo
    @next = @next - 1
  end
end
